7  Limited dispersal among patches

To confirm if there was low movement of individuals among patches in our system, we compared the maximum flight distance of the winged ant queens of each species in our dataset with the mean nearest-neighbour distances between islands in our landscapes.

We inferred the maximum flight distance of the ant queens using a linear regression between the head width of winged ant queens and their maximum flight distance estimated by Helms 2018: maximum flight distance = 2.03 * log10(winged queen head width) + 2.37. We used head width measurements of winged queens available on AntWiki and calculated the community-weighted mean of the inferred maximum flight distance for each landscape. We then calculated the mean nearest-neighbour distance between sampled islands and their nearest islands in each landscape to compare between-patch distances to the maximum flight distance estimates.

7.1.1 Loading packages and importing data

library(BAT)
library(dplyr)

# Importing data
rm(list = ls()); gc()
set.seed(13)

comm <- read.csv("ants_comm.csv", header = T)
env <- read.csv("sampled_lands.csv", header = T)

complete <- merge(comm, env, by = "point_ID")
complete <- complete[,c(1,146:154,2:145)]

comm_landscape <- complete[,-c(1,2,6)] %>%
  group_by(landscape) %>%
  summarise(
    across(c(number_patches, forest_cover, long, lat), first),
    across(c(patch_area), mean), #mean patch size by landscape
    across(
      where(is.numeric) & 
        !all_of(c("number_patches", "forest_cover", "long", "lat", "patch_area")),
      sum
    ),
    .groups = "drop"
  )

comm_landscape <- comm_landscape[, -c(2:6)]
remove(comm)
remove(complete)
remove(env)

comm_landscape <- as.data.frame(comm_landscape)
rownames(comm_landscape) <- comm_landscape$landscape
comm_landscape <- comm_landscape[,-1]

traits <- read.csv("ant_traits.csv")
setdiff(traits$especie, colnames(comm_landscape)[1:144])
traits$winged_queen <- as.factor(traits$winged_queen)

winged_queens <- subset(traits, winged_queen == "Yes")
remove(traits)

I used the regression equation from Helms (2018) to estimate the flight distance of winged ant species and then calculated the community-weighted mean (CWM) estimated flight distance for each landscape.

# S1

patches <- get_patches(S1_raster, directions = 8, class = 1, return_raster = TRUE)
patches_raster <- patches$layer_1$class_1
point_ants$point_ID2 <- terra::extract(patches_raster, point_ants)[,2]

enn_patches <- lsm_p_enn(S1_raster, directions = 8)
enn_patches <- subset(enn_patches, enn_patches$class == 1)
pontos_enn <- merge(point_ants, enn_patches, by.x = "point_ID2", by.y = "id")
media_paisagem <- mean(pontos_enn$value)
mean_dist_landscapes <- data.frame(
  id = "S1",
  mean_dist = media_paisagem
